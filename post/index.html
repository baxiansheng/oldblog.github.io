use test;

create table student
(Sno char(9) primary KEY);

INSERT INTO student values('201901001');

alter table student add State int;

drop trigger if exists Student_State;                      //创建触发器
DELIMITER ;;
create trigger Student_State before insert on student 
for each row begin
	set NEW.State = 1 ;
end
;;
DELIMITER ; 

INSERT INTO student values('201901003',null);

update student set State = 0 where Sno = '201901001';

insert into student values('201901004',null),('201901005',null);     //测试对于for each row  是否能一次性对插入的多条数据使触发器起作用

1.爬取3D打印网论坛中的帖子

通过分析帖子的html页面的标签，使用python中的BeautifulSoup库进行爬取，并将爬取到的数据写入到mysql数据库中。

  • 关于数据库的操作代码如下：

    db = pymysql.connect("localhost", "root", "woaini", "sdprint", charset='utf8') //打开数据库连接
cursor = db.cursor() //使用cursor()方法获取操作游标
sql = "INSERT INTO sdprint(class,title, author,time, content)VALUES (%s,%s,%s, %s, %s);" //构造sql语句
params = (sdprint["类别"],sdprint["标题"], sdprint["作者"],sdprint["时间"], sdprint["正文"]) //参数
try:
cursor.execute(sql,params) //执行sql语句
db.commit() //提交
print('success')
except:
db.rollback()
print('error')
db.close() //关闭数据库连接

  • 关于BS使用实例代码如下：

    f = open(filename, 'r') //把.html文件打开
html = f.read()
f.close()
bs = BeautifulSoup(html, "html.parser") //bs实例化
div = bs.find("div",{"class":"middle"}) //找到类别为middle的div，存储在div变量中

  • python字典类型知识，直接上代码

    content = {} //定义一个名为content的字典
content['姓名']='张三' //添加姓名条目张三
